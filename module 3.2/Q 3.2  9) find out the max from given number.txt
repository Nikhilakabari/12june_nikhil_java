3.2 9)Write a program to find out the max from given number

ans:
#include<stdio.h>

int main()
{   
    int num = 5672889,max = 0,n;

    while(num > 0)
    {
        n = num%10;
        if(n > max)
        {
            max = n;
        }
        num /= 10;
    }
    printf("max num from given no is: %d\n",max);
    
    return 0;
}
