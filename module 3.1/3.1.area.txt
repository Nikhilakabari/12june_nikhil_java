area of rectangle

#include <stdio.h>
#include<math.h>

int main() {
    int l,b;
    float area;
    printf("enter the l : ");
    scanf("%d",&l);
    printf("\nenter the b: ");
    scanf("%d",&b);
    area = l*b;
    printf("area of rectangle is :%f",area);
    
    return 0;
}


area of circle

#include <stdio.h>
#include<math.h>
#define pi 3.14
int main() {
    int r;
    float area;
    printf("enter the r : ");
    scanf("%d",&r);
    area = pi * pow((float)r,2);
    printf("area of circle is :%f",area);
    
    return 0;
}
