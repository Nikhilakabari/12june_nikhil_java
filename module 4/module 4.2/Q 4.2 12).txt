Write a program to calculate the area of circle, rectangle and triangle using 
Function Overloading 
 Rectangle: Area * breadth 
 Triangle: ½ *Area* breadth 
 Circle: Pi * Area *Area


ans:

#include <iostream>


using namespace std;

class z
{
	public:
		int rec(int a,int b)
		{
			return a*b;
		}
		int tri(int a,int b)
		{
			return 0.5*a*b;
		}
		int cir(int a)
		{
			return 3.14*a*a;
		}
};

int main()
{
	z z1;
	cout<<"area of rectangle is :"<<endl;
	cout<<z1.rec(8,4)<<endl;
	cout<<"area of triagnle is:"<<endl;
	cout<<z1.tri(9,10)<<endl;
	cout<<"area of circle is :"<<endl;
	cout<<z1.cir(56)<<endl;
}
